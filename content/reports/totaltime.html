center_vars <- function(df_in){

  df_in$c.memory_pcu <- df_in$memory_pcu - mean(df_in$memory_pcu)

  df_in$c.rating <- df_in$rating - mean(df_in$rating)

  df_in$c.NP_freq <- df_in$NP_freq - mean(df_in$NP_freq)

  df_in$c.freq_guipuzkoa <- df_in$freq_guipuzkoa - mean(df_in$freq_guipuzkoa)

  df_in$c.NP_mean_age <- df_in$NP_mean_age - mean(df_in$NP_mean_age)

  df_in$c.Nchar <- df_in$Nchar - mean(df_in$Nchar)

  df_in$c.verb_oind_freq <- df_in$verb_oind_freq - mean(df_in$verb_oind_freq)

  df_in$c.control_preferences <- df_in$control_preferences - mean(df_in$control_preferences)

  return(df_in)
}
